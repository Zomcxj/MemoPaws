PySide6 无边框窗口 + 系统原生动画教程

本教程介绍如何使用 PySide6 创建无边框自定义窗口，并集成 Windows/macOS 系统原生动画效果。通过 ctypes 调用系统 API，实现窗口阴影、圆角、Aero Snap 等原生体验。

环境准备

确保已安装以下依赖： 

# 安装 PySide6
pip install PySide6

# Windows 用户还需要 comtypes（可选）
pip install comtypes

创建无边框窗口

首先创建一个基础的无边框窗口类。关键在于设置 Qt.FramelessWindowHint 标志： 

import sys
from PySide6.QtWidgets import (QApplication, QMainWindow, QWidget,
                            QVBoxLayout, QPushButton)
from PySide6.QtCore import Qt


class FramelessWindow(QMainWindow):
    """无边框主窗口"""

    def __init__(self):
        super().__init__()
        self.setWindowTitle("Frameless Window")
        self.resize(800, 600)

        # 设置无边框 + 允许任务栏显示
        self.setWindowFlags(
            Qt.WindowType.FramelessWindowHint
            | Qt.WindowType.Window
        )

        # 设置半透明背景（用于圆角裁剪）
        self.setAttribute(
            Qt.WidgetAttribute.WA_TranslucentBackground
        )

        self._init_ui()

    def _init_ui(self):
        central = QWidget()
        layout = QVBoxLayout(central)
        btn = QPushButton("Close")
        btn.clicked.connect(self.close)
        layout.addWidget(btn)
        self.setCentralWidget(central)

Windows 原生阴影与动画

在 Windows 10/11 上，可以通过 ctypes 调用 DwmExtendFrameIntoClientArea 和 SetWindowAttribute 来实现系统级别的窗口阴影和圆角效果： 

import ctypes
from ctypes import wintypes, c_int, byref, sizeof


def enable_shadow(hwnd):
    """为窗口句柄启用 DWM 阴影"""
    class MARGINS(ctypes.Structure):
        _fields_ = [
            ("cxLeftWidth", c_int),
            ("cxRightWidth", c_int),
            ("cyTopHeight", c_int),
            ("cyBottomHeight", c_int),
        ]

    margins = MARGINS(-1, -1, -1, -1)
    ctypes.windll.dwmapi.DwmExtendFrameIntoClientArea(
        hwnd, byref(margins)
    )


def enable_round_corners(hwnd):
    """Windows 11 圆角窗口"""
    DWMWA_WINDOW_CORNER_PREFERENCE = 33
    DWMWCP_ROUND = 2
    ctypes.windll.dwmapi.DwmSetWindowAttribute(
        hwnd,
        DWMWA_WINDOW_CORNER_PREFERENCE,
        byref(c_int(DWMWCP_ROUND)),
        sizeof(c_int)
    )

拖拽移动窗口

由于无边框窗口没有标题栏，需要自己实现拖拽功能。重写 mousePressEvent、mouseMoveEvent 和 mouseReleaseEvent： 

def mousePressEvent(self, event):
    if event.button() == Qt.MouseButton.LeftButton:
        self._drag_pos = event.globalPosition().toPoint()
        event.accept()

def mouseMoveEvent(self, event):
    if hasattr(self, '_drag_pos'):
        delta = event.globalPosition().toPoint() - self._drag_pos
        self.move(self.pos() + delta)
        self._drag_pos = event.globalPosition().toPoint()
        event.accept()

def mouseReleaseEvent(self, event):
    if hasattr(self, '_drag_pos'):
        del self._drag_pos
        event.accept()

注意事项

  • Windows 上需要调用 DwmExtendFrameIntoClientArea 才能获得系统阴影效果
  • macOS 上无边框窗口默认带有系统阴影，无需额外处理
  • 使用 WA_TranslucentBackground 时，子控件也需要正确处理透明度
  • Linux (Wayland) 下无边框窗口行为可能因 compositor 不同而有所差异
  • 建议为自定义标题栏添加双击最大化、右键菜单等交互

完整示例

以下是一个完整的无边框窗口示例，集成了阴影、圆角和拖拽功能： 

import sys
import platform
from PySide6.QtWidgets import QApplication, QMainWindow
from PySide6.QtCore import Qt


class ModernWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowFlags(Qt.WindowType.FramelessWindowHint)
        self.setAttribute(Qt.WidgetAttribute.WA_TranslucentBackground)
        self.resize(900, 650)
        self.show()

        # Windows: 启用原生阴影和圆角
        if platform.system() == "Windows":
            hwnd = int(self.winId())
            enable_shadow(hwnd)
            enable_round_corners(hwnd)


if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = ModernWindow()
    sys.exit(app.exec())
